# Установка скрипта проверки для нескольких баз данных

Пошаговая инструкция по установке системы проверки бэкапов для нескольких баз данных MySQL.

## 📋 Что будет установлено

```
/usr/local/bin/
├── mysql_backup_verify.sh          ← Основной скрипт проверки
└── mysql_backup_verify_all.sh      ← Wrapper для нескольких БД

/var/log/
├── mysql_backup_verify_all.log     ← Общий лог wrapper
├── mysql_backup_verify_zabbix.log  ← Лог для Zabbix
├── mysql_backup_verify_nextcloud.log
└── mysql_backup_verify_wordpress.log

/data/nfs/backup/mysql-vm/
├── zabbix.sql                      ← Бэкапы от Webmin
├── nextcloud.sql
├── wordpress.sql
├── archive_zabbix/                 ← Проверенные копии
│   ├── zabbix_*_verified.sql
│   └── zabbix_*_verified.sql.md5
├── archive_nextcloud/
│   ├── nextcloud_*_verified.sql
│   └── nextcloud_*_verified.sql.md5
└── archive_wordpress/
    ├── wordpress_*_verified.sql
    └── wordpress_*_verified.sql.md5
```

## 🚀 Установка

### Шаг 1: Скачивание скриптов на NFS сервер

```bash
# SSH на Debian (NFS сервер)
ssh root@your-debian-server

# Скачайте основной скрипт
sudo nano /usr/local/bin/mysql_backup_verify.sh
# Вставьте содержимое из артефакта "mysql_backup_verify.sh"
# Сохраните: Ctrl+O, Enter, Ctrl+X

# Скачайте wrapper скрипт
sudo nano /usr/local/bin/mysql_backup_verify_all.sh
# Вставьте содержимое из артефакта "mysql_backup_verify_all.sh"
# Сохраните: Ctrl+O, Enter, Ctrl+X
```

### Шаг 2: Установка прав

```bash
# Сделайте скрипты исполняемыми
sudo chmod +x /usr/local/bin/mysql_backup_verify.sh
sudo chmod +x /usr/local/bin/mysql_backup_verify_all.sh

# Проверьте права
ls -lh /usr/local/bin/mysql_backup_verify*.sh
```

### Шаг 3: Настройка wrapper скрипта

```bash
# Откройте wrapper скрипт для редактирования
sudo nano /usr/local/bin/mysql_backup_verify_all.sh
```

**Измените следующие параметры:**

```bash
# 1. Укажите путь к директории бэкапов (ЛОКАЛЬНЫЙ путь на NFS сервере!)
MAIN_BACKUP_DIR="/data/nfs/backup/mysql-vm"

# 2. Укажите ваш email
EMAIL_ALERT="admin@example.com"

# 3. ВАЖНО! Укажите ваши базы данных
DATABASES=(
    "zabbix"      # Если файл zabbix.sql
    "nextcloud"   # Если файл nextcloud.sql
    "wordpress"   # Если файл wordpress.sql
    # Добавьте свои базы данных
)

# 4. Настройте параметры (опционально)
MAX_FILE_AGE=180        # 3 часа
RETENTION_DAYS=30       # Хранить 30 дней
```

**Сохраните файл:** Ctrl+O, Enter, Ctrl+X

### Шаг 4: Проверка конфигурации

```bash
# Проверьте что директория существует
ls -la /data/nfs/backup/mysql-vm/

# Проверьте что бэкапы там есть
ls -lh /data/nfs/backup/mysql-vm/*.sql

# Должны увидеть файлы типа:
# -rw-r--r-- 1 root root 50M Feb  6 02:15 zabbix.sql
# -rw-r--r-- 1 root root 15M Feb  6 02:15 nextcloud.sql
```

### Шаг 5: Создание директорий для архивов

```bash
# Создайте директории для каждой БД (wrapper создаст автоматически, но можно и вручную)
sudo mkdir -p /data/nfs/backup/mysql-vm/archive_zabbix
sudo mkdir -p /data/nfs/backup/mysql-vm/archive_nextcloud
sudo mkdir -p /data/nfs/backup/mysql-vm/archive_wordpress

# Установите права
sudo chmod 755 /data/nfs/backup/mysql-vm/archive_*
```

### Шаг 6: Тестовый запуск

```bash
# Запустите wrapper скрипт вручную
sudo /usr/local/bin/mysql_backup_verify_all.sh
```

### Шаг 7: Проверка логов

```bash
# Общий лог wrapper скрипта
sudo tail -50 /var/log/mysql_backup_verify_all.log

# Логи для каждой БД
sudo tail -50 /var/log/mysql_backup_verify_zabbix.log
sudo tail -50 /var/log/mysql_backup_verify_nextcloud.log
sudo tail -50 /var/log/mysql_backup_verify_wordpress.log
```

### Шаг 8: Проверка созданных копий

```bash
# Проверьте что копии создались
ls -lh /data/nfs/backup/mysql-vm/archive_zabbix/
ls -lh /data/nfs/backup/mysql-vm/archive_nextcloud/
ls -lh /data/nfs/backup/mysql-vm/archive_wordpress/

# Должны увидеть файлы типа:
# -rw-r--r-- 1 root root 50M Feb  6 15:30 zabbix_20260206_153045_verified.sql
# -rw-r--r-- 1 root root  33 Feb  6 15:30 zabbix_20260206_153045_verified.sql.md5

# Проверьте MD5
md5sum -c /data/nfs/backup/mysql-vm/archive_zabbix/*.md5
```

### Шаг 9: Настройка автозапуска через cron

```bash
# Редактируйте crontab
sudo crontab -e
```

**Добавьте строку:**

```bash
# Запуск проверки всех баз в 03:00 каждый день
# (если Webmin создает бэкапы в 02:00)
0 3 * * * /usr/local/bin/mysql_backup_verify_all.sh
```

**Или с записью в лог:**

```bash
# С выводом в лог (рекомендуется)
0 3 * * * /usr/local/bin/mysql_backup_verify_all.sh >> /var/log/mysql_backup_verify_all_cron.log 2>&1
```

**Сохраните:** Ctrl+O, Enter, Ctrl+X (в nano) или :wq (в vim)

### Шаг 10: Проверка cron задачи

```bash
# Просмотр установленных задач
sudo crontab -l

# Должны увидеть:
# 0 3 * * * /usr/local/bin/mysql_backup_verify_all.sh
```

## 🔍 Проверка работы через день

### После первого автоматического запуска:

```bash
# 1. Проверьте общий лог
sudo tail -100 /var/log/mysql_backup_verify_all.log

# Должны увидеть итоги проверки:
# ==========================================
# ИТОГИ ПРОВЕРКИ:
# Успешно проверено: 3
# Провалено проверок: 0
# ==========================================
# ✓ ВСЕ ПРОВЕРКИ ЗАВЕРШЕНЫ УСПЕШНО

# 2. Проверьте логи отдельных БД
for db in zabbix nextcloud wordpress; do
    echo "=== $db ==="
    tail -5 /var/log/mysql_backup_verify_${db}.log
done

# 3. Проверьте архивы
for db in zabbix nextcloud wordpress; do
    echo "=== $db ==="
    ls -lh /data/nfs/backup/mysql-vm/archive_${db}/ | tail -3
done

# 4. Общая статистика
echo "Всего проверенных копий:"
find /data/nfs/backup/mysql-vm/archive_*/ -name "*_verified.sql" | wc -l
```

## 🎯 Добавление новой базы данных

Если Webmin начал создавать бэкап новой БД:

```bash
# 1. Откройте wrapper скрипт
sudo nano /usr/local/bin/mysql_backup_verify_all.sh

# 2. Добавьте новую БД в массив DATABASES
DATABASES=(
    "zabbix"
    "nextcloud"
    "wordpress"
    "new_database"    ← Добавьте сюда
)

# 3. Сохраните: Ctrl+O, Enter, Ctrl+X

# 4. Тестовый запуск
sudo /usr/local/bin/mysql_backup_verify_all.sh

# 5. Проверьте что создалась директория и лог
ls -la /data/nfs/backup/mysql-vm/archive_new_database/
ls -la /var/log/mysql_backup_verify_new_database.log
```

## 🗑️ Удаление базы данных из проверки

```bash
# 1. Откройте wrapper скрипт
sudo nano /usr/local/bin/mysql_backup_verify_all.sh

# 2. Удалите или закомментируйте БД
DATABASES=(
    "zabbix"
    "nextcloud"
    # "wordpress"    ← Закомментировали
)

# 3. Опционально: удалите старые архивы и логи
sudo rm -rf /data/nfs/backup/mysql-vm/archive_wordpress/
sudo rm -f /var/log/mysql_backup_verify_wordpress.log
```

## 📊 Мониторинг и отчеты

### Ежедневная проверка статуса

```bash
# Создайте alias для быстрого просмотра
echo 'alias backup-status="tail -30 /var/log/mysql_backup_verify_all.log | grep -A 10 ИТОГИ"' >> ~/.bashrc
source ~/.bashrc

# Теперь можно просто запустить:
backup-status
```

### Скрипт еженедельного отчета

```bash
sudo nano /usr/local/bin/backup_weekly_report.sh
```

```bash
#!/bin/bash

echo "======================================"
echo "MySQL Backup Weekly Report"
echo "Period: $(date -d '7 days ago' +%Y-%m-%d) - $(date +%Y-%m-%d)"
echo "======================================"
echo ""

for db in zabbix nextcloud wordpress; do
    echo "Database: $db"
    echo "  Successful verifications: $(grep "успешно" /var/log/mysql_backup_verify_${db}.log | grep "$(date +%Y-%m)" | wc -l)"
    echo "  Failed verifications: $(grep "провалена" /var/log/mysql_backup_verify_${db}.log | grep "$(date +%Y-%m)" | wc -l)"
    echo "  Archive size: $(du -sh /data/nfs/backup/mysql-vm/archive_${db}/ | cut -f1)"
    echo "  Copies count: $(ls -1 /data/nfs/backup/mysql-vm/archive_${db}/*.sql 2>/dev/null | wc -l)"
    echo ""
done

echo "======================================"
```

```bash
sudo chmod +x /usr/local/bin/backup_weekly_report.sh

# Запуск
sudo /usr/local/bin/backup_weekly_report.sh

# Добавьте в cron для еженедельной отправки (каждый понедельник в 09:00)
0 9 * * 1 /usr/local/bin/backup_weekly_report.sh | mail -s "Weekly Backup Report" admin@example.com
```
